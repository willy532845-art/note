這題看到她說要轉動，所以單純用stack或queue都不太夠
乾脆用雙向佇列deque吧
然後因為C++沒有PY的rotate函式，為求方便就自己打包

```
#include<bits/stdc++.h>
using namespace std;

vector<deque<char>> v;//存雙向佇列，因為題目轉動的部分最好操作
deque<char> a;


char rotate (deque<char> &b) { //從front跑到back
    char t = b.front();
    b.pop_front();
    b.push_back(t);
    return t;
}

char rotate_bar (deque<char> &b) { //反相
    char t = b.back();
    b.pop_back();
    b.push_front(t);
    return t;
}

int main() {
    ios::sync_with_stdio(false); 
    cin.tie(nullptr); //IO優化
    int n, m, k, e, cnt = 0; //e是輸入參數，做參數暫存
    char t;
    cin >> m >> n >> k; 

    for (int i = 0; i< m; i++) {
        for (int j = 0; j<n ; j++) {
            cin >> t;
            a.push_back(t);
        }
        v.push_back(a);
        a.clear();
    }

    unordered_map<char,int> mp;
    for (int i = 0; i< k; i++) { //局數控制
        for (int j = 0; j<m ; j++) { //輪盤數控制
            int step; 
            cin >> step;
            step = (step % n + n) % n; //對step取mod，可以避免大數字的模擬，也可以讓轉動步數必為正數，比鰾單純

            while (step--) {
                retate_bar(v[j]);
            }
        }
        for (int y = 0; y < n ; y++) { //格子數
            mp.clear();
            int mx = 0;
            for (int x = 0; x<m ; x++) { //輪盤數
                mp[v[x][y]]++;
            }
            for (auto p : mp) {
                if (p.second > mx) {
                    mx = p.second;
                }
            }
            cnt += mx; // 把這一格出現次數最多的次數加進總分
        }
    }
    cout << cnt <<"\n";
}
```
