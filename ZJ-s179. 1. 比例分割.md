
這題用前綴和+對前綴和陣列的二分搜可以直覺得解開
比較需要注意的實作細節:
1.移項恆等式可以避免浮點數誤差
2.0-based和1-based的轉換

```
#include <bits/stdc++.h>
using namespace std;

int n, m, t, k; //t是輸入參數
vector<long long> b; //前綴和陣列

int tt(int l, int r, long long abar, long long bbar){ //這裡二分搜
    long long total_seg = b[r - 1] - (l - 2 >= 0 ? b[l - 2] : 0);
    long long target = abar * total_seg; 
    int low = l - 1, high = r - 1, ans = r - 1;
    while (low <= high) {
        int mid = low + (high - low) / 2;
        long long current_seg = b[mid] - (l - 2 >= 0 ? b[l - 2] : 0);
        
        if (current_seg * (abar + bbar) >= target) {
            ans = mid;
            high = mid - 1; // 太大
        } else {
            low = mid + 1; // 不夠大
        }
    }
    return ans + 1;
}
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL); 
    
    int l, r, qa, qb; 
    cin >> n >> m; 

    cin >> t;
    b.push_back(t); //做第一項，以利後面做前綴和
    for (int i = 1; i < n; i++){ // 修正為分號
        cin >> t; 
        b.push_back(b[i-1] + t); 
    }
    while (m--){
        cin >> l >> r >> qa >> qb; 
        cout << tt(l, r, qa, qb) << "\n";
    }
    return 0;
}
```
